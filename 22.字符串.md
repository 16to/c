# 字符串
字符串是字符的序列，通常用于表示文本。在C语言中，字符串以空字符（'\0'）结尾。

```c
#include <stdio.h>
#include <string.h>

int main() {
    char str1[] = "Hello";
    char str2[] = "World";
    char str3[100];

    // 字符串拼接
    strcpy(str3, str1); // 复制str1到str3
    strcat(str3, " ");  // 拼接空格
    strcat(str3, str2); // 拼接str2
    printf("%s\n", str3);

    // 字符串复制
    char str4[100];
    strcpy(str4, str1);
    printf("%s\n", str4);

    // 字符串长度
    printf("Length of str1: %lu\n", strlen(str1));

    // 字符串比较
    if (strcmp(str1, str2) == 0) {
        printf("str1 and str2 are equal\n");
    } else {
        printf("str1 and str2 are not equal\n");
    }

    // 字符串查找
    char *ptr = strstr(str3, "World");
    if (ptr != NULL) {
        printf("Found 'World' in str3 at position: %ld\n", ptr - str3);
    } else {
        printf("'World' not found in str3\n");
    }

    return 0;
}
```
