# 输入输出
## printf函数
```c
#include <stdio.h>

int main() {
    int num = 10;
    float pi = 3.14;
    char ch = 'A';
    printf("Integer: %d\n", num); // 输出整数
    printf("Float: %.2f\n", pi);   // 输出浮点数，保留两位小数
    printf("Character: %c\n", ch); // 输出字符
    return 0;
}
```

## scanf函数
```c
#include <stdio.h> 

int main() {
    int num;
    float pi;
    char ch;
    printf("Enter an integer: ");
    scanf("%d", &num); // 输入整数
    printf("Enter a float: ");
    scanf("%f", &pi);  // 输入浮点数
    printf("Enter a character: ");
    scanf(" %c", &ch); // 输入字符，注意前面的空格
    printf("You entered: %d, %.2f, %c\n", num, pi, ch);
    return 0;
}
